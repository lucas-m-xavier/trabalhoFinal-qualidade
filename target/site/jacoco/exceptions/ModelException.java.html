package exceptions;

/**
 *
 * @author gabriel
 */
public class ModelException extends RuntimeException{
    
    private final String messageError;
    
    public ModelException(String messageError){
        super();
        this.messageError = messageError;
    }
    
    @Override
    public String getMessage(){
        return messageError;
    }
    
}
